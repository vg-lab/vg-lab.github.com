211  NeuronMorphologyPtr
212  AscReaderTemplated< ASC_READER_TEMPLATE_CLASS_NAMES >::readMorphology(
213  const std::string& fileName, bool reposition_ )
214  {
215 
216  inFile.open( fileName, std::ios::in );
217 
219  if ( ( inFile.rdstate( ) & std::ifstream::failbit ) != 0 )
220  {
221  Log::log( std::string( "Error opening file: " ) + fileName,
222  LOG_LEVEL_WARNING );
223  return nullptr;
224 
225  }
226 
227  std::string lineString;
228 
229  NsolVector<NodePtr> repositionNodes;
230  NeuronMorphologyPtr neuronMorphology( new NEURONMORPHOLOGY( new SOMA ) );
231 
232  nodeId = 0;
233 
235  while ( std::getline( inFile, lineString ) )
236  {
238  _eraseComent( lineString );
240  std::transform( lineString.begin( ), lineString.end( ),
241  lineString.begin( ), ::tolower );
242 
244  if ( std::regex_match( lineString,
245  std::regex( ".*\\(\\s*dendrite\\s*\\)\\s*[\\r]?" ) ) )
246  {
251  DendritePtr basalP = new DENDRITE( Dendrite::BASAL );
252  neuronMorphology->addNeurite( basalP );
253  basalP->morphology( neuronMorphology );
254  _ReadNeurite( basalP, &repositionNodes, reposition_ );
255 
256  }
258  else if ( std::regex_match( lineString,
259  std::regex( ".*\\(\\s*apical\\s*\\)\\s*[\\r]?" ) ) )
260  {
265  DendritePtr apicalP = new DENDRITE( Dendrite::APICAL );
266  neuronMorphology->addNeurite( apicalP );
267  apicalP->morphology( neuronMorphology );
268  _ReadNeurite( apicalP, &repositionNodes, reposition_ );
269 
270  }
276  else if ( std::regex_match( lineString,
277  std::regex( ".*\\(\\s*cellbody\\s*\\)\\s*[\\r]?" ) ) )
278  {
279  _ReadSoma( neuronMorphology, &repositionNodes, reposition_ );
280 
281  }
283  else if ( std::regex_match( lineString,
284  std::regex( ".*\\(\\s*axon\\s*\\)\\s*[\\r]?" ) ) )
285  {
290  AxonPtr axon = new AXON( );
291  neuronMorphology->addNeurite( axon );
292  axon->morphology( neuronMorphology );
293  _ReadNeurite( axon, &repositionNodes, reposition_ );
294  }
296 
297  }
298 
303  if ( reposition_ )
304  {
305  Vec3f center = neuronMorphology->soma( )->center( );
306 
307  for ( auto& node : repositionNodes )
308  {
309  node->point( node->point( ) - center );
310  }
311 
312  neuronMorphology->soma( )->center( Vec3f( 0.0f, 0.0f, 0.0f ) );
313 
314  }
315  return neuronMorphology;
316 
317  }

320  template < ASC_READER_TEMPLATE_CLASSES > void
321  AscReaderTemplated< ASC_READER_TEMPLATE_CLASS_NAMES >::_ReadNeurite(
322  NeuritePtr neuritePointer,
323  NsolVector<NodePtr>* repositionNodes_,
324  bool reposition_ )
325  {
326  std::stack<NeuronMorphologySectionPtr> parentSections;
327  NeuronMorphologySectionPtr currentSection = nullptr;
328  std::string lineString;
329  int level = 1;
330 
331  currentSection = NeuronMorphologySectionPtr( new NEURONMORPHOLOGYSECTION );
332  neuritePointer->firstSection( currentSection );
333 
334 
335  NeuronMorphologySectionPtr parentSection = nullptr;
336  bool firstNode = true;
337 
338 
340  while ( level > 0 && std::getline( inFile, lineString ) )
341  {
343  _eraseComent( lineString );
344 
346  if( std::regex_match( lineString, std::regex( REGEX_DATA_LINE ) ) )
347  {
349  NodePtr node = _parseDataLine( lineString );
350 
355  if( reposition_ )
356  {
357  repositionNodes_->push_back( node );
358  }
359 
361  if ( firstNode )
362  {
363  currentSection->firstNode( node );
364  firstNode = false;
365  }
367  else
368  {
369  currentSection->addNode( node );
370  }
371  }
372 
377  else if ( std::regex_match( lineString, std::regex( "\\s*\\|\\s*" ) ) )
378  {
379 
384  currentSection =
385  NeuronMorphologySectionPtr( new NEURONMORPHOLOGYSECTION );
386  currentSection->neurite( neuritePointer );
387  currentSection->parent( parentSection );
388  parentSection->addChild( currentSection );
389  neuritePointer->_addBranchCount( 1 );
390 
391  }
393  else if ( std::regex_match( lineString, std::regex( REGEX_SPINE_LINE ) ) )
394  {
395  Log::log( std::string( "Spines still not implemented: " ) +
396  lineString, LOG_LEVEL_VERBOSE );
397 
398  }
399  else
400  {
401  int bracketCount = _countBrackets( lineString );
402  level += bracketCount;
403 
405  if( bracketCount < 0 && level > 1 )
406  {
407  parentSections.pop( );
408  parentSection = parentSections.top( );
409 
410  }
411  else if( bracketCount > 0 )
412  {
413  std::transform( lineString.begin( ), lineString.end( ),
414  lineString.begin( ), ::tolower );
416  if ( std::regex_match( lineString,
417  std::regex( "\\s*\\(\\s*dot\\s*[\\r]?" ) ) )
418  {
419  bracketCount = 0;
420 
421  while ( bracketCount == 0 && std::getline( inFile, lineString ) )
422  {
423  _eraseComent( lineString );
424  bracketCount = _countBrackets( lineString );
425 
426  }
427  level += bracketCount;
428 
429  }
430  else
431  {
437  parentSections.push( currentSection );
438  neuritePointer->_addBifurcationCount( 1 );
439  neuritePointer->_addBranchCount( 1 );
440 
441  parentSection = currentSection;
442  currentSection = new NEURONMORPHOLOGYSECTION( );
443  currentSection->neurite( neuritePointer );
444  currentSection->parent( parentSection );
445  parentSection->addChild( currentSection );
446 
447  }
448  }
449  }
450  }
