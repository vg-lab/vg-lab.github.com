220  template < SWC_READER_TEMPLATE_CLASSES >
221  NeuronMorphologyPtr
222  SwcReaderTemplated< SWC_READER_TEMPLATE_CLASS_NAMES >::readMorphology(
223  const std::string& fileName, bool reposition_ )
224  {
225  std::ifstream inFile;
226  inFile.open( fileName, std::ios::in );
227 
229  if ( ( inFile.rdstate( ) & std::ifstream::failbit ) != 0 )
230  {
231  Log::log( std::string( "Error opening file: " ) + fileName, LOG_LEVEL_WARNING );
232 
233  return nullptr;
234 
235  }
236 
237  std::string lineString;
238 
239  NsolVector< NodePtr > repositionNodes;
240  NeuronMorphologyPtr neuronMorphology( new NEURONMORPHOLOGY( new SOMA ) );
241 
242  std::map<unsigned int, TSwcLine> lines;
243 
244  int lineCount = 0;
245 
247  while ( std::getline( inFile, lineString ) )
248  {
249  ++lineCount;
250 
252  if ( lineString[lineString.find_first_not_of( " \r\t" )] != '#' )
253  {
255  unsigned int fields =
256  1 + ( unsigned int )std::count_if( lineString.begin( ), lineString.end( ),
257  []( unsigned char c )
258  {
259  return std::isspace( c );
260  }
261  );
262 
263  if ( fields < 7 )
264  {
265  Log::log( std::string( "Skipping line " ) +
266  std::to_string( lineCount ) +
267  std::string( " in file " ) +
268  fileName + std::string( ": \"" ) + lineString +
269  std::string( "\" ( not enough fields found )" ),
270  LOG_LEVEL_WARNING );
271  }
272  else
273  {
275  std::istringstream iss( lineString );
276  TSwcLine swcLineString;
277 
278  bool failed = false;
279  iss >> swcLineString.id;
280  if ( iss.fail( ) )
281  failed = true;
282  iss >> swcLineString.type;
283  if ( iss.fail( ) )
284  failed = true;
285  iss >> swcLineString.xyz[0]
286  >> swcLineString.xyz[1]
287  >> swcLineString.xyz[2];
288  if ( iss.fail( ) )
289  failed = true;
290  iss >> swcLineString.radius;
291  if ( iss.fail( ) )
292  failed = true;
293  iss >> swcLineString.parent;
294  if ( iss.fail( ) )
295  failed = true;
296 
297  if ( failed )
298  {
299  Log::log( std::string( "Skipping line " ) +
300  std::to_string( lineCount ) +
301  std::string( " in file " ) +
302  fileName + std::string( ": \"" ) + lineString +
303  std::string( "\" ( line format not recognised )" ),
304  LOG_LEVEL_WARNING );
305  } else {
306  lines[swcLineString.id] = swcLineString;
307  }
308 
309  }
310  }
311  }
312 
313  inFile.close( );
314 
316  for ( const auto& line : lines )
317  {
318  if ( line.second.parent != -1 )
319  lines[line.second.parent].children.push_back( line.first );
320  }
321 
322  std::vector<unsigned int> somaChildren;
323  std::map<unsigned int, NodePtr > nodeSomaPtr;
324 
326  for ( const auto& line : lines )
327  {
328  if ( line.second.type == SWC_SOMA )
329  {
330  NodePtr node(
331  new NODE( line.second.xyz, line.second.id, line.second.radius ) );
332 
337  if ( reposition_ )
338  repositionNodes.push_back( node );
339 
340  neuronMorphology->soma( )->addNode( node );
341 
342  nodeSomaPtr[line.second.id] = node;
343 
345  for ( const unsigned int child : line.second.children )
346  if ( lines[child].type != SWC_SOMA )
347  somaChildren.push_back( child );
348 
349  }
350  }
351 
353  for ( const unsigned int firstNodeId : somaChildren )
354  {
355 
356  switch ( lines[firstNodeId].type ) {
357  case SWC_BASAL:
358  {
359  DendritePtr basalP = new DENDRITE( Dendrite::BASAL );
360  neuronMorphology->addNeurite( basalP );
361  basalP->morphology( neuronMorphology );
362  _ReadNeurite( basalP, lines, firstNodeId,
363  &repositionNodes, reposition_ );
364  }
365  break;
366 
367  case SWC_APICAL:
368  {
369  DendritePtr apicalP = new DENDRITE( Dendrite::APICAL );
370  neuronMorphology->addNeurite( apicalP );
371  apicalP->morphology( neuronMorphology );
372  _ReadNeurite( apicalP, lines, firstNodeId,
373  &repositionNodes, reposition_ );
374  }
375  break;
376 
377  case SWC_AXON:
378  {
379  AxonPtr axonP = new AXON();
380  neuronMorphology->addNeurite( axonP );
381  axonP->morphology( neuronMorphology );
382  _ReadNeurite( axonP, lines, firstNodeId,
383  &repositionNodes, reposition_ );
384  }
385  break;
386 
387  default:
388  Log::log( std::string( "Unexpected line type value in line " ) +
389  std::to_string( firstNodeId ), LOG_LEVEL_WARNING);
390 
391  }
392  }
393 
399  if ( reposition_ )
400  {
401  Vec3f center = neuronMorphology->soma( )->center( );
402 
403  for ( auto& node : repositionNodes )
404  {
405  node->point( node->point( ) - center );
406  }
407 
408  neuronMorphology->soma( )->center( Vec3f( 0.0f, 0.0f, 0.0f ) );
409 
410  }
411 
412  return neuronMorphology;
413 
414  }

417  template < SWC_READER_TEMPLATE_CLASSES > void
418  SwcReaderTemplated< SWC_READER_TEMPLATE_CLASS_NAMES >::_ReadNeurite(
419  NeuritePtr neuritePointer,
420  const std::map<unsigned int, TSwcLine>& lines,
421  unsigned int initId,
422  NsolVector<NodePtr>* nodes_,
423  bool reposition_ )
424  {
425  std::stack<TReadNeuriteStackElem> sectionFirstNodes;
426  NeuronMorphologySectionPtr sectionPointer = nullptr;
427 
428  sectionPointer = NeuronMorphologySectionPtr( new NEURONMORPHOLOGYSECTION );
429 
430  sectionPointer->neurite( neuritePointer );
431  sectionPointer->parent( nullptr );
432 
433  NodePtr node;
435  TSwcLine lineElem = lines.at( initId );
436  node = new NODE( lineElem.xyz, initId, lineElem.radius );
437  sectionPointer->firstNode( node );
439  _ReadSection( neuritePointer, sectionPointer, node, &sectionFirstNodes, lines, nodes_, reposition_ );
440  neuritePointer->firstSection( sectionPointer );
441 
443  while ( !sectionFirstNodes.empty( ) )
444  {
445  unsigned int id = sectionFirstNodes.top( ).id;
446  NeuronMorphologySectionPtr parentSection = sectionFirstNodes.top( ).parent;
447  sectionPointer = NeuronMorphologySectionPtr( new NEURONMORPHOLOGYSECTION );
448  sectionPointer->neurite( neuritePointer );
449  sectionPointer->parent( parentSection );
450  sectionFirstNodes.pop( );
451 
453  lineElem = lines.at( id );
454  node = new NODE( lineElem.xyz, id, lineElem.radius );
455  sectionPointer->addNode( node ); //
456 
457  parentSection->addChild( sectionPointer );
458 
460  _ReadSection( neuritePointer, sectionPointer, node, &sectionFirstNodes,
461  lines, nodes_, reposition_ );
462 
463  }
464  }

466  template < SWC_READER_TEMPLATE_CLASSES > void
467  SwcReaderTemplated< SWC_READER_TEMPLATE_CLASS_NAMES >::_ReadSection(
468  NeuritePtr neuritePointer,
469  NeuronMorphologySectionPtr sectionPointer,
470  NodePtr nodePointer,
471  std::stack<TReadNeuriteStackElem>* sectionFirstNodes,
472  const std::map<unsigned int, TSwcLine>& lines,
473  NsolVector<NodePtr>* nodes_,
474  bool reposition_ )
475  {
476  unsigned int nodeId = ( unsigned int ) nodePointer->id();
478  if ( reposition_ )
479  nodes_->push_back( nodePointer );
480 
482  TSwcLine lineElem = lines.at( nodeId );
483  while ( lineElem.children.size( ) == 1 )
484  {
485  nodeId = lineElem.children[0];
486  lineElem = lines.at( nodeId );
487  nodePointer = new NODE( lineElem.xyz, nodeId, lineElem.radius );
488 
490  if ( reposition_ )
491  nodes_->push_back( nodePointer );
492 
493  sectionPointer->addNode( nodePointer );
494  }
495 
500  if ( lineElem.children.size( ) > 1 )
501  {
503  neuritePointer->_addBranchCount( ( unsigned int ) lineElem.children.size( ) );
505  neuritePointer->_addBifurcationCount( 1 );
506 
508  for ( const auto& it : lineElem.children )
509  {
510  TReadNeuriteStackElem tmpStackElem = { it, sectionPointer };
511  sectionFirstNodes->push( tmpStackElem );
512  }
513 
514  }
515  }
