158  brain::Circuit brainCircuit( *_blueConfig );
159  brion::GIDSet gidSetBrain = brainCircuit.getGIDs( _target );
160  const brain::Synapses& brainSynapses = brainCircuit.
161  getAfferentSynapses( gidSetBrain,
162  brain::SynapsePrefetch::all );
163 
164  if( _circuit.synapses().size() != brainSynapses.size() )
165  {
166  Log::log("Not exist correspondence of the neurons data "
167  "between NSOL and BRION", LOG_LEVEL_WARNING );
168  return;
169  }
170 
171  for( unsigned int i = 0; i < _circuit.synapses().size(); ++i )
172  {
173  const brain::Synapse& brainSynapse = brainSynapses[ i ];
174  MorphologySynapsePtr synapse = dynamic_cast<MorphologySynapsePtr>
175  ( _circuit.synapses()[i] );
176 
177  if( synapse == nullptr )
178  {
179  Log::log("Inconsistent type of synapse.", LOG_LEVEL_WARNING );
180  return;
181  }
182 
183  vmml::Vector3f brainPreSynPos = brainSynapse.
184  getPresynapticSurfacePosition();
185  vmml::Vector3f brainPostSynPos = brainSynapse.
186  getPostsynapticSurfacePosition();
187 
188  synapse->preSynapticSurfacePosition( Vec3f( brainPreSynPos.x(),
189  brainPreSynPos.y(),
190  brainPreSynPos.z()));
191 
192  synapse->postSynapticSurfacePosition( Vec3f( brainPostSynPos.x(),
193  brainPostSynPos.y(),
194  brainPostSynPos.z()));
195 
196  std::unordered_map< unsigned int, NeuronPtr >::const_iterator
197  nitPre = _neurons.find( synapse->preSynapticNeuron( ));
198  std::unordered_map< unsigned int, NeuronPtr >::const_iterator
199  nitPost = _neurons.find( synapse->postSynapticNeuron( ));
200 
201  if (( nitPre == _neurons.end( ))||
202  ( nitPost == _neurons.end( )))
203  continue;
204 
205  NeuronPtr preSynapticNeuron = nitPre->second;
206  NeuronPtr postSynapticNeuron = nitPost->second;
207 
208  // Computing pre-synaptic section..
209  bool found = false;
210  NeuronMorphologySectionPtr preSynSection = nullptr;
211  unsigned int brainIDSection = brainSynapse.getPresynapticSectionID();
212 
213  Neurites neuritesPre = preSynapticNeuron->morphology()->neurites();
214  NSOL_FOREACH( neuriteIt, neuritesPre )
215  {
216  if( found ) break;
217 
218  const NeuritePtr neurite = (*neuriteIt);
219  Sections sections = neurite->sections();
220 
221  NSOL_FOREACH( sectionIt, sections )
222  {
223  const NeuronMorphologySectionPtr section =
224  dynamic_cast< NeuronMorphologySectionPtr>( *(sectionIt));
225  if( section->id() == brainIDSection )
226  {
227  found = true;
228  preSynSection = section;
229 
230  break;
231  }
232  }// for all sections of one neurite
233  }// for all neurites
234 
235 
236  // Computing post-synaptic section..
237  found = false;
238 
239  NeuronMorphologySectionPtr postSynSection = nullptr;
240  brainIDSection = brainSynapse.getPostsynapticSectionID();
241 
242  Neurites neuritesPost = postSynapticNeuron->morphology()->neurites();
243  NSOL_FOREACH( neuriteIt, neuritesPost )
244  {
245  if( found ) break;
246 
247  const NeuritePtr neurite = (*neuriteIt);
248  Sections sections = neurite->sections();
249 
250  NSOL_FOREACH( sectionIt, sections )
251  {
252  const NeuronMorphologySectionPtr section =
253  dynamic_cast< NeuronMorphologySectionPtr >( *( sectionIt ));
254 
255  if( section->id() == brainIDSection )
256  {
257  found = true;
258  postSynSection = section;
259 
260  break;
261  }
262  }// for all sections of one neurite
263  }// for all neurites
264 
265  synapse->preSynapticSection( preSynSection );
266  synapse->postSynapticSection( postSynSection );
267 
268  }// all synapses
