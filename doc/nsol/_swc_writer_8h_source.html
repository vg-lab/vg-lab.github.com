63  bool writeMorphology(const std::string& fileName, NeuronMorphologyPtr morphology)
64  {
65 
66  if (!morphology)
67  {
68  std::cerr << "Error neuron " << std::endl;
69  return false;
70  }
71 
72  std::ofstream outFile;
73  outFile.open(fileName, std::ios::out);
74 
75  //Opening file checking
76  if ((outFile.rdstate() & std::ofstream::failbit) != 0)
77  {
78  std::cerr << "Error opening file " << fileName << std::endl;
79  return false;
80  }
81 
82  int parent = -1;
83 
84 // TODO: remove this initial line restriction
85  outFile << "#File exporter" << std::endl;
86 
87  //Writing soma
88  for (NsolVector<NodePtr>::iterator it =
89  morphology->soma()->nodes().begin();
90  it != morphology->soma()->nodes().end(); ++it)
91  {
92  outFile << " "
93  << (*it)->id()
94  << " "
95  << int( SWC_SOMA )
96  << " "
97  << (*it)->point()[0]
98  << " "
99  << (*it)->point()[1]
100  << " "
101  << (*it)->point()[2]
102  << " "
103  << (*it)->radius()
104  << " "
105  << parent
106  << std::endl;
107 
108  parent = (*it)->id();
109  }
110 
111  Neurites neurites = morphology->neurites();
112  unsigned int type;
113 
114  //Writing neurites
115  for (Neurites::iterator it = neurites.begin();
116  it != neurites.end(); ++it)
117  {
118  if (NSOL_DYNAMIC_CAST( Dendrite, *it ))
119  {
120  if (NSOL_DYNAMIC_CAST( Dendrite, *it )->dendriteType() == Dendrite::APICAL)
121  type = SWC_APICAL;
122  else if (NSOL_DYNAMIC_CAST( Dendrite, *it )->dendriteType() == Dendrite::BASAL)
123  type = SWC_DENDRITE;
124  }
125  else if (NSOL_DYNAMIC_CAST( Axon, *it ))
126  type = SWC_AXON;
127 
128  NeuronMorphologySectionPtr fSection = (*it)->firstSection();
129 
130  std::stack<SectionPtr> sPS;
131  sPS.push(fSection);
132 
133  std::map<unsigned int, NodePtr> nodePtrMap;
134  std::map<unsigned int, int> nodeParentId;
135 
136  nodePtrMap[ fSection->firstNode( )->id( ) ] = fSection->firstNode( );
137  nodeParentId[ fSection->firstNode( )->id( ) ] = 1;
138 
139  while (!sPS.empty())
140  {
141  NeuronMorphologySectionPtr lS =
142  dynamic_cast< NeuronMorphologySectionPtr >( sPS.top( ));
143  sPS.pop();
144 
145  NSOL_FOREACH( node, lS->nodes() )
146  {
147  if ( (*node) != lS->firstNode( ) )
148  {
149  nodePtrMap[( *node )->id( )] = *node;
150  nodeParentId[( *node )->id( )] = parent;
151  }
152  parent = ( *node )->id( );
153  }
154 
155  if (lS->children().size() > 0)
156  for (unsigned int i = 0; i < lS->children().size(); ++i)
157  sPS.push(lS->children()[i]);
158  }
159 
160  for (std::map<unsigned int, NodePtr>::iterator nodeIt =
161  nodePtrMap.begin(); nodeIt != nodePtrMap.end(); ++nodeIt)
162  {
163 
164  outFile << " "
165  << nodeIt->second->id()
166  << " "
167  << type
168  << " "
169  << nodeIt->second->point()[0]
170  << " "
171  << nodeIt->second->point()[1]
172  << " "
173  << nodeIt->second->point()[2]
174  << " "
175  << nodeIt->second->radius()
176  << " "
177  << nodeParentId[nodeIt->second->id()]
178  << std::endl;
179 
180  parent = nodeParentId[nodeIt->second->id()];
181  }
182 
183  }
184 
185  outFile.close();
186 
187  return true;
188 
189  }
